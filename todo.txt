移除沒有使用的package
https://dev.to/manitej/remove-unused-npm-modules-in-less-than-30-seconds-4g8k
npm install -g depcheck

npx depcheck | sed '/Missing/q' | grep '\*' | cut -c 3- | while read my_package; do
  npm uninstall "$my_package"
done
